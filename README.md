            <script>
                (function () {
                    const section = document.getElementById("stats-strip");
                    if (!section) return;

                    const counters = Array.from(section.querySelectorAll(".countup"));
                    const animateCounter = (el, duration) => {
                        const target = Number(el.dataset.target || "0");
                        const suffix = el.dataset.suffix || "";
                        const start = performance.now();
                        const from = 0;

                        const step = (now) => {
                            const t = Math.min(1, (now - start) / duration);
                            const eased = 1 - Math.pow(1 - t, 3);
                            const value = Math.round(from + (target - from) * eased);
                            el.textContent = String(value) + suffix;
                            if (t < 1) requestAnimationFrame(step);
                        };

                        requestAnimationFrame(step);
                    };

                    let ran = false;
                    const io = new IntersectionObserver(
                        (entries) => {
                            const entry = entries[0];
                            if (!entry || !entry.isIntersecting || ran) return;
                            ran = true;
                            counters.forEach((c, i) => animateCounter(c, 1100 + i * 120));
                        },
                        { threshold: 0.4 }
                    );

                    io.observe(section);
                })();

    (function () {
      const items = Array.from(document.querySelectorAll(".faq-item"));
      const panels = Array.from(document.querySelectorAll(".faq-panel"));

      panels.forEach((p) => {
        p.style.transition = "max-height 320ms ease, opacity 240ms ease";
      });

      const setOpen = (item, open) => {
        const btn = item.querySelector(".faq-trigger");
        const panel = item.querySelector(".faq-panel");
        const toggleWrap = item.querySelector(".faq-toggle");
        const toggleIcon = item.querySelector(".faq-toggle-icon");

        if (!btn || !panel || !toggleWrap || !toggleIcon) return;

        btn.setAttribute("aria-expanded", open ? "true" : "false");

        if (open) {
          panel.style.opacity = "1";
          panel.style.maxHeight = panel.scrollHeight + "px";
        } else {
          panel.style.opacity = "0";
          panel.style.maxHeight = "0px";
        }

        toggleIcon.classList.toggle("fa-plus", !open);
        toggleIcon.classList.toggle("fa-minus", open);

        toggleWrap.classList.toggle("bg-slate-100", !open);
        toggleWrap.classList.toggle("text-slate-700", !open);
        toggleWrap.classList.toggle("bg-[#0030AC]", open);
        toggleWrap.classList.toggle("text-white", open);
      };

      items.forEach((item) => {
        const btn = item.querySelector(".faq-trigger");
        if (!btn) return;

        btn.addEventListener("click", () => {
          const isOpen = btn.getAttribute("aria-expanded") === "true";
          items.forEach((it) => setOpen(it, false));
          if (!isOpen) setOpen(item, true);
        });
      });

      const init = () => {
        let hasOpen = false;
        items.forEach((item) => {
          const btn = item.querySelector(".faq-trigger");
          if (!btn) return;
          const open = btn.getAttribute("aria-expanded") === "true";
          if (open) hasOpen = true;
          setOpen(item, open);
        });
        if (!hasOpen && items[0]) setOpen(items[0], true);
      };

      window.addEventListener("load", init);
      window.addEventListener("resize", () => {
        const openBtn = document.querySelector(".faq-trigger[aria-expanded='true']");
        if (!openBtn) return;
        const item = openBtn.closest(".faq-item");
        const panel = item ? item.querySelector(".faq-panel") : null;
        if (panel) panel.style.maxHeight = panel.scrollHeight + "px";
      });
    })();
